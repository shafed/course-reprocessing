\section*{Введение: Произведения векторов и их применение в IT}

В этой лекции мы изучим три операции над векторами: \textbf{скалярное},
\textbf{векторное} и \textbf{смешанное произведения}. В отличие от
сложения векторов, эти операции учитывают геометрическую взаимосвязь
векторов — углы между ними, ориентацию в пространстве, объёмы.

Для студента IT-направления эти операции являются фундаментом целого ряда
практических областей.

\begin{remarkIT}{Компьютерная графика и освещение}{cg_intro}
  \begin{itemize}
    \item \textbf{Скалярное произведение:} Модель освещения Ламберта
      вычисляет яркость грани как $I = I_0 \cdot \max((\vec{n}, \vec{l}), 0)$,
      где $\vec{n}$ — нормаль к полигону, $\vec{l}$ — направление на источник
      света. Одно скалярное произведение на вершину в вершинном шейдере.
    \item \textbf{Векторное произведение:} Нормаль к полигону вычисляется как
      $\vec{n} = [\overrightarrow{AB}, \overrightarrow{AC}]$ по двум рёбрам
      треугольника. Ориентация нормали (направление «лицевой» стороны)
      определяется порядком операндов.
    \item \textbf{Смешанное произведение:} Проверка, находится ли точка внутри
      выпуклого многогранника, сводится к знакам смешанных произведений.
  \end{itemize}
\end{remarkIT}

\begin{remarkIT}{Машинное обучение и наука о данных}{ml_intro}
  \begin{itemize}
    \item \textbf{Скалярное произведение:} В методе опорных векторов (SVM)
      разделяющая гиперплоскость $(\vec{w}, \vec{x}) + b = 0$ и расстояние
      от точки до неё $d = |(\vec{w}, \vec{x}_0) + b| / |\vec{w}|$ —
      всё это скалярные произведения.
    \item \textbf{Косинусное сходство} $\cos\alpha = (\va, \vb)/(|\va||\vb|)$
      — стандартная метрика близости документов и векторных представлений
      слов (word embeddings) в NLP.
  \end{itemize}
\end{remarkIT}

\begin{remarkIT}{Робототехника и навигация}{robo_intro}
  Момент силы $\vec{M} = [\vec{r}, \vec{F}]$ — векторное произведение плеча
  на силу — используется контроллером промышленного манипулятора для расчёта
  крутящего момента в каждом суставе. Смешанное произведение позволяет
  быстро оценить объём рабочего пространства манипулятора и обнаружить
  вырожденные конфигурации ($V = 0$, когда оси суставов компланарны).
\end{remarkIT}

\begin{definition}{Угол между векторами}{angle_def}
  \textbf{Углом} $\alpha$ между векторами $\va$ и $\vb$ называется наименьший
  угол $AOB$. При этом $0 \leq \alpha \leq \pi$.
\end{definition}

\begin{definition}{Скалярное произведение}{scalar_prod_def}
  \textbf{Скалярным произведением} векторов $\va$ и $\vb$ называется число
  \begin{equation}
    (\va,\,\vb) = |\va|\,|\vb|\cos\alpha, \qquad
    \text{где } \alpha = \widehat{(\va,\vb)}. \tag{8.1}
  \end{equation}
\end{definition}

\noindent
Из определения непосредственно следует связь скалярного произведения
с проекцией:
\[
  (\va,\vb) = |\va|\,\operatorname{пр}_{\va}\vb
  = |\vb|\,\operatorname{пр}_{\vb}\va,
\]
поскольку $\operatorname{пр}_{\vb}\va = |\va|\cos\alpha$, откуда
$\operatorname{пр}_{\vb}\va = \dfrac{(\va,\vb)}{|\vb|}$.

Таким образом, скалярное произведение измеряет,
\textit{насколько один вектор «смотрит» в направлении другого}.

\begin{remarkIT}{Модель освещения Ламберта}{lambert}
  В реальном рендеринге (OpenGL, Vulkan) яркость грани вычисляется как
  \[
    I = I_0 \cdot \max\!\bigl((\vec{n},\,\vec{l}),\;0\bigr),
  \]
  где $\vec{n}$ --- единичная нормаль к полигону, $\vec{l}$ ---
  единичный вектор на источник света, $I_0$ --- интенсивность.
  Вся операция сводится к одному скалярному произведению на вершину
  в вершинном шейдере. Если $(\vec{n},\vec{l})\le 0$, грань
  повёрнута от света и затемняется.
\end{remarkIT}

\medskip

\textbf{\textit{Свойства скалярного произведения}}

\begin{enumerate}
  \item \textbf{Коммутативность:} $(\va,\vb) = (\vb,\va)$.

    \begin{myproof}
      $(\va,\vb) = |\va|\,|\vb|\cos\alpha = |\vb|\,|\va|\cos\alpha =
      (\vb,\va)$.\hfill
    \end{myproof}

  \item \textbf{Дистрибутивность:}
    $(\va + \vb,\,\vc) = (\va,\vc) + (\vb,\vc)$.

    \begin{myproof}
      $(\va + \vb,\,\vc) = |\vc|\,\operatorname{пр}_{\vc}(\va + \vb)
      = |\vc|\!\left(\operatorname{пр}_{\vc}\va +
      \operatorname{пр}_{\vc}\vb\right) =
      |\vc|\,\operatorname{пр}_{\vc}\va +
      |\vc|\,\operatorname{пр}_{\vc}\vb = (\va,\vc) + (\vb,\vc)$.
    \end{myproof}

  \item \textbf{Однородность:} $(\lambda\va,\vb) = \lambda(\va,\vb)$.

    \begin{myproof}
      При $\lambda>0$: вектор $\lambda\va$ сонаправлен с $\va$, угол
      с $\vb$ тот же $\alpha$, а $|\lambda\va|=\lambda|\va|$, поэтому
      $(\lambda\va,\vb)=\lambda|\va|\,|\vb|\cos\alpha=\lambda(\va,\vb)$.
      При $\lambda<0$: вектор противоположен, угол равен $\pi-\alpha$,
      $\cos(\pi-\alpha)=-\cos\alpha$, а $|\lambda\va|=|\lambda|\,|\va|$;
      знаки компенсируются и результат тот же.
    \end{myproof}

  \item \textbf{Критерий перпендикулярности:}
    $(\va,\vb) = 0 \;\Leftrightarrow\; \va \perp \vb$
    при $\va \neq \vec{0},\; \vb \neq \vec{0}$.

    \begin{myproof}
      $(\va,\vb)=|\va|\,|\vb|\cos\alpha=0
      \;\Leftrightarrow\;\cos\alpha=0
      \;\Leftrightarrow\;\alpha=\tfrac{\pi}{2}
      \;\Leftrightarrow\;\va\perp\vb$.
    \end{myproof}

  \item \textbf{Связь с длиной:}
    $(\va,\va) = |\va|^2 \;\Rightarrow\; |\va| = \sqrt{(\va,\va)}$.

    \begin{myproof}
      При $\va=\vb$ имеем $\alpha=0$, значит
      $(\va,\va)=|\va|^2\cos 0=|\va|^2$.
    \end{myproof}
\end{enumerate}

\begin{remarkIT}{Критерий перпендикулярности в машинном обучении}{svm_perp}
  В методе опорных векторов (SVM) разделяющая гиперплоскость имеет
  вид $(\vec{w},\vec{x})+b=0$. Класс объекта $\vec{x}$ определяется
  знаком $(\vec{w},\vec{x})+b$: свойство~4 гарантирует, что $\vec{w}$
  перпендикулярен гиперплоскости, а расстояние от точки $\vec{x}_0$
  до неё равно
  $d = \dfrac{|(\vec{w},\vec{x}_0)+b|}{|\vec{w}|}$
  --- это \textit{отступ} (margin), который SVM максимизирует
  при обучении.
\end{remarkIT}

\medskip

\textbf{\textit{Координатная запись скалярного произведения}}

Пусть $\va = (x_1, y_1, z_1)$ и $\vb = (x_2, y_2, z_2)$ в
ортонормированном базисе $\{\vi,\vj,\vk\}$, где
$(\vi,\vi)=(\vj,\vj)=(\vk,\vk)=1$ и
$(\vi,\vj)=(\vi,\vk)=(\vj,\vk)=0$.
Раскрывая по свойствам билинейности:
\begin{align*}
  (\va, \vb)
  &= (x_1\vi + y_1\vj + z_1\vk,\; x_2\vi + y_2\vj + z_2\vk) \\[6pt]
  &= x_1 x_2(\vi,\vi) + x_1 y_2(\vi,\vj) + x_1 z_2(\vi,\vk) \\[4pt]
  &\quad + y_1 x_2(\vj,\vi) + y_1 y_2(\vj,\vj) + y_1 z_2(\vj,\vk) \\[4pt]
  &\quad + z_1 x_2(\vk,\vi) + z_1 y_2(\vk,\vj) + z_1 z_2(\vk,\vk) \\[6pt]
  &= x_1 x_2 + y_1 y_2 + z_1 z_2.
\end{align*}
Таким образом,
\begin{equation}
  \boxed{(\va, \vb) = x_1 x_2 + y_1 y_2 + z_1 z_2.} \tag{8.2}
\end{equation}

\textbf{\textit{Угол между векторами в координатах}}

Из определения скалярного произведения получаем формулу для угла:
\[
  \cos\alpha = \frac{(\va,\vb)}{|\va|\,|\vb|},
  \qquad \alpha = \widehat{(\va,\vb)},
\]
и в координатах:
\begin{equation}
  \cos\alpha =
  \frac{x_1 x_2 + y_1 y_2 + z_1 z_2}
  {\sqrt{x_1^2+y_1^2+z_1^2}\;\sqrt{x_2^2+y_2^2+z_2^2}}. \tag{8.3}
\end{equation}

\begin{remarkIT}{Угол между нормалями в компьютерном зрении}{cv_normals}
  Алгоритмы сегментации облаков точек (LiDAR, RGB-D) определяют
  принадлежность двух соседних точек одной плоской грани по углу между
  нормалями: если $\alpha < \varepsilon$ (порог $5^\circ$--$15^\circ$),
  точки объединяются в один сегмент. Все вычисления сводятся к одному
  применению формулы (8.3).
\end{remarkIT}

\textbf{\textit{Примеры}}

\begin{exampleBox}{Перпендикулярность граней куба}{sc1}
  Грани единичного куба имеют внешние нормали
  $\vec{n}_1 = (1,\,0,\,0)$ и $\vec{n}_2 = (0,\,1,\,0)$.
  Найти угол между ними и убедиться, что грани перпендикулярны.

  \textbf{Решение.}
  \[
    (\vec{n}_1,\vec{n}_2) = 1{\cdot}0 + 0{\cdot}1 + 0{\cdot}0 = 0.
  \]
  \[
    \cos\alpha = \frac{0}{1\cdot 1} = 0
    \;\Rightarrow\; \alpha = \frac{\pi}{2}.
  \]
  По критерию~4 нулевое скалярное произведение равносильно
  перпендикулярности, что соответствует геометрии куба.
  В алгоритмах детектирования столкновений (collision detection)
  нормали двух граней с нулевым скалярным произведением указывают на
  взаимно перпендикулярные поверхности --- особый случай, требующий
  отдельной ветки обработки.
\end{exampleBox}

\begin{exampleBox}{Проекция скорости звена манипулятора}{sc2}
  Контроллер сустава робота-манипулятора работает с вектором скорости
  звена $\vec{v} = (3,\,-2,\,6)$ (м/с). Ось вращения сустава задана
  единичным вектором $\vec{e} = (0,\,0,\,1)$.
  Найти проекцию $\vec{v}$ на ось вращения и угол между ними.

  \textbf{Решение.}

  Длина: $|\vec{v}|=\sqrt{9+4+36}=\sqrt{49}=7$, $|\vec{e}|=1$.
  \[
    \operatorname{пр}_{\vec{e}}\vec{v}
    = \frac{(\vec{v},\vec{e})}{|\vec{e}|}
    = \frac{3{\cdot}0+(-2){\cdot}0+6{\cdot}1}{1}
    = 6 \text{ (м/с).}
  \]
  \[
    \cos\alpha = \frac{6}{7\cdot 1} = \frac{6}{7}
    \;\Rightarrow\;
    \alpha = \arccos\frac{6}{7} \approx 31{,}0^\circ.
  \]
  Контроллер использует проекцию $6$ м/с как компоненту скорости,
  непосредственно вращающую звено; поперечная компонента нагружает раму.
\end{exampleBox}

\begin{exampleBox}{Расстояние от точки до гиперплоскости SVM}{sc3}
  Классификатор SVM задан параметрами $\vec{w}=(3,\,4)$, $b=-5$.
  Найти расстояние от точки $\vec{x}_0=(1,\,2)$ до разделяющей
  гиперплоскости и определить её класс.

  \textbf{Решение.}

  $|\vec{w}|=\sqrt{9+16}=5$.
  \[
    (\vec{w},\vec{x}_0)+b = 3{\cdot}1+4{\cdot}2+(-5) = 3+8-5 = 6.
  \]
  \[
    d = \frac{|(\vec{w},\vec{x}_0)+b|}{|\vec{w}|}
    = \frac{6}{5} = 1{,}2.
  \]
  Так как $(\vec{w},\vec{x}_0)+b=6>0$, точка принадлежит
  положительному классу. Расстояние $d=1{,}2$ до границы говорит
  о высокой уверенности классификатора.
\end{exampleBox}

\begin{exampleBox}{Угол между диагональю куба и ребром}{sc4}
  Найти угол между главной диагональю куба и его ребром.
  Пусть ребро направлено вдоль $\vec{a}=(1,\,0,\,0)$,
  а главная диагональ --- $\vec{d}=(1,\,1,\,1)$.

  \textbf{Решение.}

  $|\vec{a}|=1$, $|\vec{d}|=\sqrt{3}$.
  \[
    \cos\alpha = \frac{(\vec{a},\vec{d})}{|\vec{a}|\,|\vec{d}|}
    = \frac{1}{1\cdot\sqrt{3}}
    = \frac{1}{\sqrt{3}}
    \;\Rightarrow\;
    \alpha = \arccos\frac{1}{\sqrt{3}} \approx 54{,}7^\circ.
  \]
  Этот угол --- основа \textit{изометрической проекции}: при повороте
  сцены на $54{,}7^\circ$ от плоскости экрана диагональ куба
  смотрит прямо на наблюдателя, и все три ребра проецируются одинаково.
  Именно такой вид используется в аксонометрических видах CAD-систем.
\end{exampleBox}

\section*{Векторное произведение векторов}

\begin{definition}{Компланарные векторы}{coplanar_def}
  Векторы называются \textbf{компланарными}, если они лежат в одной
  плоскости или в параллельных плоскостях.
\end{definition}

\begin{definition}{Упорядоченная тройка векторов}{ordered_triple_def}
  Тройку векторов, приведённых к одному началу, называют
  \textbf{упорядоченной}, если известно, какой из них первый, какой второй
  и какой третий.
\end{definition}

\begin{definition}{Ориентация тройки векторов}{orientation_def}
  Упорядоченная тройка некомпланарных векторов называется \textbf{правой},
  если из конца третьего вектора кратчайший поворот от первого вектора ко
  второму виден против часовой стрелки. В противном случае тройка называется
  \textbf{левой}.
\end{definition}

\noindent
\textbf{Замечание.} Стандартный базис $\{\vi,\vj,\vk\}$ декартовой
системы координат образует правую тройку. Смена ориентации (переход
к левой тройке) соответствует зеркальному отражению пространства.

\begin{remarkIT}{Ориентация в компьютерной графике}{orientation_cg}
  В OpenGL используется правая система координат, в DirectX --- левая.
  Это означает, что порядок обхода вершин треугольника (против или по
  часовой стрелке) определяет, какая сторона полигона считается лицевой
  (front face). Ошибка в ориентации тройки --- классическая причина
  «невидимых» моделей: видеодрайвер отбрасывает все тыльные грани
  (back-face culling), и объект оказывается прозрачным изнутри.
\end{remarkIT}

\begin{definition}{Векторное произведение}{cross_prod_def}
  \textbf{Векторным произведением} векторов $\va$ и $\vb$ называется
  вектор $\vc = [\va, \vb]$, удовлетворяющий условиям:
  \begin{enumerate}
    \item $\vc \perp \va$ и $\vc \perp \vb$;
    \item $|\vc| = |\va|\,|\vb|\sin\alpha$, где $\alpha = \widehat{(\va,\vb)}$:
      \begin{equation}
        |[\va,\vb]| = |\va|\,|\vb|\sin\alpha; \tag{8.4}
      \end{equation}
    \item $(\va, \vb, \vc)$ --- правая тройка.
  \end{enumerate}
\end{definition}

\noindent
\textbf{Замечание.} Условие~2 означает, что $|\vc|$ равно площади
параллелограмма, построенного на $\va$ и $\vb$, а при $\va \parallel
\vb$ (т.е. $\sin\alpha = 0$) получаем $[\va,\vb] = \vec{0}$.

\begin{remarkIT}{Нормаль к полигону в графике}{normal_cg}
  Именно из условия~1 следует, что $[\overrightarrow{AB},
  \overrightarrow{AC}]$ перпендикулярен плоскости треугольника $ABC$.
  Видеодрайвер вычисляет нормаль каждого полигона ровно так при
  импорте геометрии: два ребра дают нормаль через одно векторное
  произведение. Ориентация нормали (вправо или влево) определяется
  порядком операндов --- следствие условия~3.
\end{remarkIT}

\textbf{\textit{Свойства векторного произведения}}

\begin{enumerate}
  \item \textbf{Антикоммутативность:} $[\va, \vb] = -[\vb, \va]$.

    \begin{myproof}
      Векторы $[\va,\vb]$ и $[\vb,\va]$ имеют одинаковые длину
      ($|\va|\,|\vb|\sin\alpha$) и перпендикулярны тем же $\va$, $\vb$.
      Единственное отличие --- ориентация: при перестановке операндов
      тройка меняет ориентацию с правой на левую, значит вектор
      направлен в противоположную сторону. Следовательно,
      $[\vb,\va] = -[\va,\vb]$.
    \end{myproof}

  \item \textbf{Дистрибутивность:}
    $[\va + \vb,\,\vc] = [\va, \vc] + [\vb, \vc]$.

    \begin{myproof}
      Доказательство геометрически сводится к тому, что проекция
      $\va + \vb$ на плоскость, перпендикулярную $\vc$, равна сумме
      проекций $\va$ и $\vb$. Затем поворот этих проекций на $90^\circ$
      и умножение на $|\vc|$ даёт линейность. Координатная проверка
      производится прямой подстановкой в формулу определителя.
    \end{myproof}

  \item \textbf{Однородность:} $[\lambda\va, \vb] = \lambda[\va, \vb]$.

    \begin{myproof}
      При $\lambda > 0$: $|\lambda\va| = \lambda|\va|$, угол с $\vb$
      тот же, ориентация не меняется, поэтому длина и направление
      результата умножаются на $\lambda$.
      При $\lambda < 0$: $|\lambda\va| = |\lambda|\,|\va|$, но вектор
      $\lambda\va$ противоположен $\va$, тройка меняет ориентацию на
      противоположную, что даёт дополнительный знак минус, итого
      $[\lambda\va,\vb] = \lambda[\va,\vb]$.
    \end{myproof}

  \item \textbf{Критерий параллельности:}
    $[\va, \vb] = \vec{0} \;\Leftrightarrow\; \va \parallel \vb$
    при $\va \neq \vec{0},\; \vb \neq \vec{0}$.

    \begin{myproof}
      $|[\va,\vb]| = |\va|\,|\vb|\sin\alpha = 0
      \;\Leftrightarrow\; \sin\alpha = 0
      \;\Leftrightarrow\; \alpha = 0 \text{ или } \pi
      \;\Leftrightarrow\; \va \parallel \vb$.
    \end{myproof}

  \item \textbf{Геометрический смысл модуля:}
    $|[\va, \vb]| = S_{\text{пар}}$, где $S_{\text{пар}}$ ---
    площадь параллелограмма на $\va$ и $\vb$.

    \begin{myproof}
      $S_{\text{пар}} = |\va|\cdot h = |\va|\cdot|\vb|\sin\alpha
      = |[\va,\vb]|$, где $h = |\vb|\sin\alpha$ --- высота.
    \end{myproof}
\end{enumerate}

\begin{remarkIT}{Момент силы в робототехнике}{torque_robot}
  Момент силы $\vec{M} = [\vec{r}, \vec{F}]$ --- это векторное
  произведение плеча $\vec{r}$ на силу $\vec{F}$. Контроллер
  промышленного манипулятора вычисляет крутящий момент в каждом
  суставе именно так, обходя кинематическую цепочку от рабочего
  органа к основанию. Свойство~1 (антикоммутативность) объясняет,
  почему порядок $[\vec{r},\vec{F}]$ и $[\vec{F},\vec{r}]$ даёт
  противоположные направления вращения.
\end{remarkIT}

\textbf{\textit{Координатная запись векторного произведения}}

Пусть $\va = (x_1, y_1, z_1)$ и $\vb = (x_2, y_2, z_2)$ в
ортонормированном базисе $\{\vi,\vj,\vk\}$. Используем свойства
векторного произведения базисных векторов:
\[
  [\vi,\vi] = [\vj,\vj] = [\vk,\vk] = \vec{0}; \quad
  [\vi,\vj] = \vk,\quad [\vj,\vk] = \vi,\quad [\vk,\vi] = \vj.
\]
Раскрывая по билинейности:
\begin{align*}
  [\va, \vb]
  &= [x_1\vi + y_1\vj + z_1\vk,\; x_2\vi + y_2\vj + z_2\vk] \\[6pt]
  &= x_1 x_2[\vi,\vi] + x_1 y_2[\vi,\vj] + x_1 z_2[\vi,\vk]
  + y_1 x_2[\vj,\vi] + y_1 y_2[\vj,\vj] + y_1 z_2[\vj,\vk] \\[4pt]
  &\quad + z_1 x_2[\vk,\vi] + z_1 y_2[\vk,\vj] + z_1 z_2[\vk,\vk] \\[6pt]
  &= x_1 y_2\vk - x_1 z_2\vj - y_1 x_2\vk + y_1 z_2\vi
  + z_1 x_2\vj - z_1 y_2\vi \\[4pt]
  &= (y_1 z_2 - z_1 y_2)\vi - (x_1 z_2 - x_2 z_1)\vj
  + (x_1 y_2 - y_1 x_2)\vk.
\end{align*}
Эту запись удобно свернуть в символический определитель:
\begin{equation}
  \boxed{[\va, \vb] =
    \begin{vmatrix} \vi & \vj & \vk \\ x_1 & y_1 & z_1 \\ x_2 & y_2 & z_2
  \end{vmatrix}.} \tag{8.5}
\end{equation}

\noindent
\textbf{Замечание.} Раскладывать определитель следует по первой строке.
Знаки кофакторов: $+\vi$, $-\vj$, $+\vk$.

\textbf{\textit{Примеры}}

\begin{exampleBox}{Нормаль к треугольнику в 3D-меше}{vp1}
  При импорте 3D-меша программа обрабатывает треугольник с вершинами
  $A(1,\,0,\,0)$, $B(2,\,0,\,0)$, $C(1,\,1,\,0)$.
  Найти единичную нормаль к этому треугольнику.

  \textbf{Решение.}
  \[
    \overrightarrow{AB} = (1,\,0,\,0), \qquad
    \overrightarrow{AC} = (0,\,1,\,0).
  \]
  \[
    [\overrightarrow{AB},\,\overrightarrow{AC}] =
    \begin{vmatrix} \vi & \vj & \vk \\ 1 & 0 & 0 \\ 0 & 1 & 0
    \end{vmatrix}
    = (0{\cdot}0 - 0{\cdot}1)\vi - (1{\cdot}0 - 0{\cdot}0)\vj
    + (1{\cdot}1 - 0{\cdot}0)\vk
    = (0,\,0,\,1).
  \]
  Длина: $|(0,0,1)| = 1$, нормаль уже единична:
  $\vec{n} = (0,\,0,\,1)$.
  Полигон лежит в плоскости $XOY$, нормаль направлена вдоль оси $Oz$
  --- результат согласуется с геометрией.
\end{exampleBox}

\begin{exampleBox}{Момент силы рычага манипулятора}{vp2}
  Рычаг сустава манипулятора расположен вдоль оси $Oz$: плечо
  $\vec{r} = (0,\,0,\,2)$ м. К концу рычага приложена сила
  $\vec{F} = (10,\,0,\,0)$ Н. Найти вектор момента силы и его модуль.

  \textbf{Решение.}
  \[
    \vec{M} = [\vec{r},\,\vec{F}] =
    \begin{vmatrix} \vi & \vj & \vk \\ 0 & 0 & 2 \\ 10 & 0 & 0
    \end{vmatrix}
    = (0{\cdot}0 - 2{\cdot}0)\vi - (0{\cdot}0 - 2{\cdot}10)\vj
    + (0{\cdot}0 - 0{\cdot}10)\vk
    = (0,\,20,\,0) \text{ Н·м.}
  \]
  $|\vec{M}| = 20$ Н·м. Момент направлен вдоль $\vj$ --- ось вращения
  сустава совпадает с $Oy$.
\end{exampleBox}

\begin{exampleBox}{Площадь треугольника}{vp3}
  Найти площадь треугольника $ABC$, если $A(0,\,0,\,0)$,
  $B(1,\,-1,\,1)$, $C(2,\,1,\,-1)$.

  \textbf{Решение.}
  \[
    \overrightarrow{AB} = (1,\,-1,\,1), \qquad
    \overrightarrow{AC} = (2,\,1,\,-1).
  \]
  \[
    [\overrightarrow{AB},\,\overrightarrow{AC}] =
    \begin{vmatrix} \vi & \vj & \vk \\ 1 & -1 & 1 \\ 2 & 1 & -1
    \end{vmatrix}
    = ((-1)(-1) - 1{\cdot}1)\vi - (1{\cdot}(-1) - 1{\cdot}2)\vj
    + (1{\cdot}1 - (-1){\cdot}2)\vk
  \]
  \[
    = (1-1)\vi - (-1-2)\vj + (1+2)\vk = (0,\,3,\,3).
  \]
  \[
    |(0,\,3,\,3)| = \sqrt{0+9+9} = 3\sqrt{2}.
  \]
  \[
    S_{\triangle ABC} = \frac{1}{2}\,|[\overrightarrow{AB},\,
    \overrightarrow{AC}]| = \frac{3\sqrt{2}}{2}.
  \]
\end{exampleBox}

\begin{exampleBox}{Критерий параллельности векторов}{vp4}
  Проверить, параллельны ли векторы $\vec{a} = (1,\,2,\,3)$ и
  $\vec{b} = (-2,\,-4,\,-6)$.

  \textbf{Решение.}
  \[
    [\vec{a},\,\vec{b}] =
    \begin{vmatrix} \vi & \vj & \vk \\ 1 & 2 & 3 \\ -2 & -4 & -6
    \end{vmatrix}
    = (2{\cdot}(-6) - 3{\cdot}(-4))\vi - (1{\cdot}(-6) - 3{\cdot}(-2))\vj
    + (1{\cdot}(-4) - 2{\cdot}(-2))\vk
  \]
  \[
    = (-12+12)\vi - (-6+6)\vj + (-4+4)\vk = \vec{0}.
  \]
  По свойству~4, $[\vec{a},\vec{b}] = \vec{0}$ означает
  $\vec{a} \parallel \vec{b}$. Действительно, $\vec{b} = -2\vec{a}$.
\end{exampleBox}

\section*{Смешанное произведение векторов}

\begin{definition}{Смешанное произведение}{mixed_prod_def}
  \textbf{Смешанным произведением} трёх векторов $\va$, $\vb$ и $\vc$
  называется скалярное произведение векторного произведения первых двух
  на третий:
  \begin{equation}
    (\va,\,\vb,\,\vc) = ([\va,\,\vb],\,\vc). \tag{8.6}
  \end{equation}
  Смешанное произведение --- это число (скаляр), не вектор.
\end{definition}

\begin{remarkIT}{Проверка компланарности в компьютерном зрении}{coplanar_cv}
  В алгоритмах 3D-реконструкции по облаку точек (Structure from Motion,
  SLAM) часто нужно проверить, лежат ли четыре точки $A$, $B$, $C$, $D$
  в одной плоскости. Критерий прост: если
  $(\overrightarrow{AB},\overrightarrow{AC},\overrightarrow{AD})=0$,
  то точки компланарны и не несут информации о глубине --- их нельзя
  использовать для триангуляции 3D-положения. Это проверяется одним
  вычислением определителя $3\times3$.
\end{remarkIT}

\begin{theorem}{Геометрический смысл смешанного произведения}{mixed_geom}
  Пусть $V$ --- объём параллелепипеда, построенного на векторах $\va$,
  $\vb$ и $\vc$. Тогда
  \[
    (\va,\,\vb,\,\vc) =
    \begin{cases}
      +V, & \text{если } (\va,\vb,\vc) \text{ --- правая тройка}; \\
      -V, & \text{если } (\va,\vb,\vc) \text{ --- левая тройка}; \\
      \phantom{+}0,  & \text{если } \va,\vb,\vc \text{ --- компланарны}.
    \end{cases}
  \]
\end{theorem}

\textbf{\textit{Свойства смешанного произведения}}

\begin{enumerate}
  \item \textbf{Циклическая симметрия:}
    $(\va,\vb,\vc) = (\vb,\vc,\va) = (\vc,\va,\vb)$.

    \begin{myproof}
      В координатной записи смешанное произведение --- это определитель
      матрицы, строки которой суть координаты $\va$, $\vb$, $\vc$.
      Циклическая перестановка строк определителя $3\times3$ производит
      чётную перестановку (два транспозиции), не меняя его знак.
    \end{myproof}

  \item \textbf{Знак при транспозиции:} перестановка любых двух
    векторов меняет знак:
    $(\va,\vb,\vc) = -(\vb,\va,\vc)$.

    \begin{myproof}
      Транспозиция двух строк определителя меняет его знак.
    \end{myproof}

  \item \textbf{Критерий компланарности:}
    $(\va,\vb,\vc) = 0 \;\Leftrightarrow\; \va,\vb,\vc$ ---
    компланарны (при ненулевых $\va,\vb,\vc$).

    \begin{myproof}
      Из теоремы: $(\va,\vb,\vc)=0 \Leftrightarrow V=0 \Leftrightarrow$
      параллелепипед вырожден $\Leftrightarrow$ векторы компланарны.
      Эквивалентно: $\det = 0 \Leftrightarrow$ строки линейно зависимы.
    \end{myproof}

  \item \textbf{Линейность по каждому аргументу:}
    $((\alpha\va+\beta\va'),\vb,\vc)
    = \alpha(\va,\vb,\vc)+\beta(\va',\vb,\vc)$.

    \begin{myproof}
      Следует из линейности определителя по строкам.
    \end{myproof}
\end{enumerate}

\begin{remarkIT}{Объём рабочего пространства в автоматизации}{workspace_robot}
  В промышленной робототехнике рабочая зона манипулятора часто
  аппроксимируется параллелепипедом. Если три вектора $\va$, $\vb$,
  $\vc$ задают рёбра этого параллелепипеда, то
  $V = |(\va,\vb,\vc)|$ --- мгновенная оценка досягаемого объёма.
  Контроллер проверяет компланарность $(\va,\vb,\vc)=0$ как
  вырожденный случай (рабочая зона сплющилась в плоскость) и
  сигнализирует о механической неисправности.
\end{remarkIT}

\textbf{\textit{Координатная запись смешанного произведения}}

Пусть $\va=(x_1,y_1,z_1)$, $\vb=(x_2,y_2,z_2)$, $\vc=(x_3,y_3,z_3)$.
Подставим координатную формулу векторного произведения и вычислим
скалярное произведение результата на $\vc$:
\begin{align*}
  (\va,\vb,\vc) = ([\va,\vb],\vc)
  &=
  \begin{vmatrix}y_1&z_1\\y_2&z_2
  \end{vmatrix}x_3
  -
  \begin{vmatrix}x_1&z_1\\x_2&z_2
  \end{vmatrix}y_3
  +
  \begin{vmatrix}x_1&y_1\\x_2&y_2
  \end{vmatrix}z_3 \\
  &=
  \begin{vmatrix}
    x_1 & y_1 & z_1 \\
    x_2 & y_2 & z_2 \\
    x_3 & y_3 & z_3
  \end{vmatrix}.
\end{align*}
Таким образом,
\begin{equation}
  \boxed{(\va,\vb,\vc) =
    \begin{vmatrix}
      x_1 & y_1 & z_1 \\
      x_2 & y_2 & z_2 \\
      x_3 & y_3 & z_3
  \end{vmatrix}.} \tag{8.7}
\end{equation}

\noindent
\textbf{Замечание.} Циклическая симметрия из свойства~1 означает
на языке определителей, что его значение не меняется при
одновременном циклическом сдвиге всех трёх строк --- этим
объясняется равенство
$(\va,\vb,\vc) = (\vb,\vc,\va) = (\vc,\va,\vb)$.
Некратная перестановка меняет знак:
$(\va,\vb,\vc) = -(\vb,\va,\vc)$.

\textbf{\textit{Примеры}}

\begin{exampleBox}{Объём рабочего параллелепипеда}{mp1}
  Рабочая зона манипулятора описана тремя ортогональными рёбрами:
  $\va=(2,\,0,\,0)$ м, $\vb=(0,\,3,\,0)$ м, $\vc=(0,\,0,\,1)$ м.
  Вычислить объём рабочего параллелепипеда.

  \textbf{Решение.}
  \[
    (\va,\vb,\vc) =
    \begin{vmatrix}
      2 & 0 & 0 \\
      0 & 3 & 0 \\
      0 & 0 & 1
    \end{vmatrix}
    = 2\cdot(3\cdot1 - 0\cdot0)
    - 0\cdot(\ldots)
    + 0\cdot(\ldots) = 6.
  \]
  $V = |(\va,\vb,\vc)| = 6$ м³. Тройка правая (определитель положителен).
\end{exampleBox}

\begin{exampleBox}{Объём пирамиды}{mp2}
  Найти объём пирамиды $ABCD$, если
  $A(3;\,-1;\,0)$, $B(0;\,-3;\,3)$, $C(2;\,1;\,-1)$, $D(3;\,2;\,4)$.

  \textbf{Решение.}
  \[
    \overrightarrow{AB} = (-3,\,-2,\,3), \quad
    \overrightarrow{AC} = (-1,\,2,\,-1), \quad
    \overrightarrow{AD} = (0,\,3,\,4).
  \]
  \[
    (\overrightarrow{AB},\overrightarrow{AC},\overrightarrow{AD}) =
    \begin{vmatrix}
      -3 & -2 & 3 \\
      -1 &  2 & -1 \\
      0 &  3 & 4
    \end{vmatrix}.
  \]
  Разложим по первому столбцу:
  \begin{align*}
    &= -3
    \begin{vmatrix}2&-1\\3&4
    \end{vmatrix}
    -(-1)
    \begin{vmatrix}-2&3\\3&4
    \end{vmatrix}
    + 0\cdot(\ldots) \\
    &= -3(8+3) + 1(-8-9) \\
    &= -33 - 17 = -50.
  \end{align*}
  Объём параллелепипеда $V_{\text{пар}} = |-50| = 50$,
  объём пирамиды:
  \[
    V_{ABCD} = \frac{1}{6}|(\overrightarrow{AB},
    \overrightarrow{AC},\overrightarrow{AD})|
    = \frac{50}{6} = \frac{25}{3}.
  \]
\end{exampleBox}

\begin{exampleBox}{Проверка компланарности четырёх точек}{mp3}
  Проверить, лежат ли точки $A(1;\,2;\,-1)$, $B(0;\,1;\,5)$,
  $C(-1;\,2;\,1)$, $D(2;\,1;\,3)$ в одной плоскости.

  \textbf{Решение.}
  \[
    \overrightarrow{AB} = (-1,\,-1,\,6), \quad
    \overrightarrow{AC} = (-2,\,0,\,2), \quad
    \overrightarrow{AD} = (1,\,-1,\,4).
  \]
  \[
    (\overrightarrow{AB},\overrightarrow{AC},\overrightarrow{AD}) =
    \begin{vmatrix}
      -1 & -1 & 6 \\
      -2 &  0 & 2 \\
      1 & -1 & 4
    \end{vmatrix}.
  \]
  Раскроем по первой строке:
  \begin{align*}
    &= -1
    \begin{vmatrix}0&2\\-1&4
    \end{vmatrix}
    -(-1)
    \begin{vmatrix}-2&2\\1&4
    \end{vmatrix}
    +6
    \begin{vmatrix}-2&0\\1&-1
    \end{vmatrix} \\
    &= -1(0+2) + 1(-8-2) + 6(2-0) \\
    &= -2 - 10 + 12 = 0.
  \end{align*}
  По свойству~3, $(\overrightarrow{AB},\overrightarrow{AC},
  \overrightarrow{AD})=0$ означает компланарность векторов, т.е.
  точки $A$, $B$, $C$, $D$ лежат в одной плоскости.

  В системах 3D-реконструкции это означает: четыре контрольные точки
  захватывают одну грань объекта и не позволяют восстановить
  трёхмерную геометрию --- нужна хотя бы одна точка вне плоскости.
\end{exampleBox}

\begin{exampleBox}{Проверка циклической симметрии}{mp4}
  Проверить свойство циклической симметрии на конкретных векторах:
  $\va=(1,\,0,\,0)$, $\vb=(0,\,1,\,0)$, $\vc=(0,\,0,\,1)$.

  \textbf{Решение.}
  \[
    (\va,\vb,\vc) =
    \begin{vmatrix}1&0&0\\0&1&0\\0&0&1
    \end{vmatrix} = 1.
  \]
  \[
    (\vb,\vc,\va) =
    \begin{vmatrix}0&1&0\\0&0&1\\1&0&0
    \end{vmatrix}
    = 0\cdot
    \begin{vmatrix}0&1\\0&0
    \end{vmatrix}
    - 1\cdot
    \begin{vmatrix}0&1\\1&0
    \end{vmatrix}
    + 0\cdot(\ldots)
    = -1\cdot(0-1) = 1.
  \]
  \[
    (\vc,\va,\vb) =
    \begin{vmatrix}0&0&1\\1&0&0\\0&1&0
    \end{vmatrix}
    = 0\cdot(\ldots)
    - 0\cdot(\ldots)
    + 1\cdot
    \begin{vmatrix}1&0\\0&1
    \end{vmatrix} = 1.
  \]
  Все три значения равны $1$ --- циклическая симметрия подтверждена.
\end{exampleBox}
